# 常用写法

## 根据列表创建 N 叉树

```python
# Definition for a binary tree node.
class Node:
    def __init__(self, val: Optional[int] = None, children: Optional[List['Node']] = None):
        self.val = val
        self.children = children

    def __repr__(self):
        return f"{self.val}"


def list_to_tree(data, node_type = -+) -> 'Node':
    link = [[node_type(data[0], children = [])]]
    index = 0
    null_flag = False
    for i in range(1, len(data)):
        if data[i] is None:
            if not null_flag or (index == (len(link[-2]) - 1)):
                link.append([])
                null_flag = True
                index = 0
            else:
                index += 1
            continue
        tmp = node_type(data[i], children = [])
        link[-2][index].children.append(tmp)
        link[-1].append(tmp)
    return link[0][0]
```

## 根据列表创建二叉树

```python
def list_to_tree(data, node_type = TreeNode) -> 'TreeNode':
    link = [[node_type(data[0])], []]
    index = 0
    count = 0
    flag = True
    for i in range(1, len(data)):
        tmp = node_type(data[i]) if data[i] is not None else None
        if flag:
            link[-2][index].left = tmp
            flag = False
        else:
            link[-2][index].right = tmp
        if tmp:
            link[-1].append(tmp)
        count += 1
        if count % 2 == 0:
            flag = True
            index += 1
        if count >= len(link[-2]) * 2:
            link.append([])
            index = 0
            count = 0
    return link[0][0]
```

## 显示二叉树的数据

```python
def printf_tree_struct(root: Optional[TreeNode]):
    data = [[root], []]
    while data[-2]:
        count = 0
        for node in data[-2]:
            if not node:
                data[-1].append(None)
                data[-1].append(None)
                continue
            if node.left:
                count += 1
            data[-1].append(node.left)
            if node.right:
                count += 1
            data[-1].append(node.right)
        if count:
            data.append([])
        else:
            break
    data.pop()
    for item in data:
        print(item)
```
